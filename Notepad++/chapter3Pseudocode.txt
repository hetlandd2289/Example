//create module
module standardPay(float hours, float wage)
	//multiply and display result
	display (hours * wage) + " is the standard pay this week."
end module

//create module
module overtimePay(float overtime, float wage, float overtimeMultiplier)
	//multiply and display result
	display (overtime * wage * overtimeMultiplier) + " is the overtime pay this week."
end module

//create module
module totalPay(float hours, float overtime, float wage, float overtimeMultiplier)
	//multiply and add, then display
	display (hours * wage) + (overtime * wage * overtimeMultiplier) + " is the total pay this week."
end module

//create module
module main()
	//declare variables
	declare float hours = 0
	declare float overtime = 0
	declare float wage = 0
	declare float overtimeMultiplier = 1.5

	//get inputs
	display "Please enter the amount of non-overtime hours worked this week: "
	input hours
	display "Please enter the amount of overtime hours worked this week: "
	input overtime
	display "Please enter the hourly wage: "
	input wage
	
	//call other modules and calculate
	call standardPay(hours, wage)
	call overtimePay(overtime, wage, overtimeMultiplier)
	call totalPay(hours, overtime, wage, overtimeMultiplier)
end module
